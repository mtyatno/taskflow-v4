# Chat / Diskusi Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Tambah halaman Diskusi (chat real-time per shared list) dengan fitur attach task, buat task baru dari chat, @mention anggota, dan notifikasi mention — semua tanpa refresh halaman.

**Architecture:** SSE (Server-Sent Events) untuk real-time push dari server ke client. Client kirim pesan via `POST`. In-memory `dict[list_id → set[Queue]]` sebagai broadcast bus (single-worker VPS). History pesan disimpan permanen di SQLite tabel `messages`. Frontend: komponen React baru (`ChatPage`, `ChatListPanel`, `ChatRoom`, `TaskAttachPopup`) ditambahkan ke single-file SPA `static/index.html`.

**Tech Stack:** FastAPI + sse-starlette (SSE), SQLite, React 18 UMD (no build step), existing `notifications` table untuk mention alerts.

  ```python
  @app.post("/api/lists/{list_id}/messages")
  async def post_message(list_id: int, req: MessageCreate, user=Depends(get_current_user)):
      uid = user["sub"]
      repo = TaskRepository(DB_PATH)
      if not repo.is_list_member_or_owner(list_id, uid):
          raise HTTPException(status_code=403, detail="Not a member of this list")
      now = datetime.now().isoformat()
      with get_db() as conn:
          # Validate task_id belongs to this list (if provided)
          if req.task_id:
              task_row = conn.execute(
                  "SELECT id FROM tasks WHERE id = ? AND list_id = ?", (req.task_id, list_id)
              ).fetchone()
              if not task_row:
                  raise HTTPException(status_code=400, detail="Task tidak ditemukan di list ini")
          # Save message
          cur = conn.execute(
              "INSERT INTO messages (list_id, user_id, content, task_id, msg_type, created_at) "
              "VALUES (?,?,?,?,?,?)",
              (list_id, uid, req.content, req.task_id, req.msg_type, now),
          )
          msg_id = cur.lastrowid
          # Fetch with joined data for broadcast
          row = conn.execute(
              """SELECT m.id, m.list_id, m.user_id, m.content, m.task_id, m.msg_type, m.created_at,
                        u.username, u.display_name,
                        t.title as task_title, t.priority as task_priority,
                        t.deadline as task_deadline, t.quadrant as task_quadrant
                 FROM messages m
                 JOIN users u ON u.id = m.user_id
                 LEFT JOIN tasks t ON t.id = m.task_id
                 WHERE m.id = ?""",
              (msg_id,),
          ).fetchone()
          msg_dict = dict(row)
          # Process @mentions
          mentions = set(re.findall(r'@(\w+)', req.content))
          if mentions:
              list_row = conn.execute(
                  "SELECT name FROM shared_lists WHERE id = ?", (list_id,)
              ).fetchone()
              list_name = list_row["name"] if list_row else f"List #{list_id}"
              sender_username = user["username"]
              for username in mentions:
                  if username == sender_username:
                      continue
                  mentioned = conn.execute(
                      "SELECT u.id FROM users u "
                      "WHERE u.username = ? AND ("
                      "  u.id IN (SELECT owner_id FROM shared_lists WHERE id = ?) "
                      "  OR u.id IN (SELECT user_id FROM list_members WHERE list_id = ?)"
                      ")",
                      (username, list_id, list_id),
                  ).fetchone()
                  if mentioned:
                      conn.execute(
                          "INSERT INTO notifications (user_id, message, is_read, list_id, task_id, created_at) "
                          "VALUES (?,?,0,?,NULL,?)",
                          (mentioned["id"],
                           f"💬 {sender_username} menyebut kamu di diskusi '{list_name}'",
                           list_id, now),
                      )
      # Broadcast to all SSE subscribers of this list
      for q in list(chat_subscribers.get(list_id, set())):
          await q.put(msg_dict)
      return msg_dict
  ```

  Langsung setelah POST messages, tambahkan:

  ```python
  @app.get("/api/lists/{list_id}/messages/stream")
  async def chat_stream(list_id: int, request: Request, user=Depends(get_current_user)):
      uid = user["sub"]
      repo = TaskRepository(DB_PATH)
      if not repo.is_list_member_or_owner(list_id, uid):
          raise HTTPException(status_code=403, detail="Not a member of this list")

      async def event_generator():
          q: asyncio.Queue = asyncio.Queue()
          chat_subscribers[list_id].add(q)
          try:
              while True:
                  if await request.is_disconnected():
                      break
                  try:
                      msg = await asyncio.wait_for(q.get(), timeout=20.0)
                      yield {"data": json.dumps(msg)}
                  except asyncio.TimeoutError:
                      # Keepalive ping agar koneksi tidak timeout
                      yield {"data": json.dumps({"type": "ping"})}
          finally:
              chat_subscribers[list_id].discard(q)

      return EventSourceResponse(event_generator())
  ```

  function ChatInputBar({ list, user, onSent, showToast, onTaskClick }) {
    const [text, setText] = useState("");
    const [sending, setSending] = useState(false);
    const [showAttach, setShowAttach] = useState(false);
    const [attachedTask, setAttachedTask] = useState(null);
    const [showTaskForm, setShowTaskForm] = useState(false);
    const [members, setMembers] = useState([]);
    const [mentionQuery, setMentionQuery] = useState(null); // null = off, string = query
    const textareaRef = useRef(null);

    // Load members for @mention autocomplete once
    useEffect(() => {
      api.get(`/api/lists/${list.id}/members/usernames`)
        .then(setMembers)
        .catch(() => {});
    }, [list.id]);

    const mentionSuggestions = mentionQuery !== null
      ? members.filter(m =>
          m.username !== user.username &&
          m.username.toLowerCase().startsWith(mentionQuery.toLowerCase())
        )
      : [];

    const handleTextChange = (e) => {
      const val = e.target.value;
      setText(val);
      // Detect @mention trigger
      const match = val.match(/@(\w*)$/);
      if (match) {
        setMentionQuery(match[1]);
      } else {
        setMentionQuery(null);
      }
    };

    const insertMention = (username) => {
      const newVal = text.replace(/@(\w*)$/, `@${username} `);
      setText(newVal);
      setMentionQuery(null);
      textareaRef.current?.focus();
    };

    const handleSend = async () => {
      if (!text.trim() && !attachedTask) return;
      setSending(true);
      try {
        const payload = {
          content: text.trim() || (attachedTask ? `📌 ${attachedTask.title}` : ""),
          task_id: attachedTask?.id || null,
          msg_type: attachedTask ? "task_attach" : "text",
        };
        const msg = await api.post(`/api/lists/${list.id}/messages`, payload);
        setText("");
        setAttachedTask(null);
        setMentionQuery(null);
        onSent(msg);
      } catch (err) { showToast(err.message, "error"); }
      finally { setSending(false); }
    };

    const handleKeyDown = (e) => {
      if (mentionSuggestions.length > 0 && e.key === "Escape") {
        setMentionQuery(null);
        return;
      }
      if (e.key === "Enter" && !e.shiftKey) {
        e.preventDefault();
        handleSend();
      }
    };

    return (
      <div className="chat-input-bar">
        {/* @mention dropdown */}
        {mentionSuggestions.length > 0 && (
          <div className="chat-mention-dropdown">
            {mentionSuggestions.map(m => (
              <div key={m.username} className="chat-mention-item" onClick={() => insertMention(m.username)}>
                <span style={{ fontWeight: 600 }}>@{m.username}</span>
                {m.display_name && m.display_name !== m.username && (
                  <span style={{ color: "var(--text-light)", marginLeft: 6, fontSize: 12 }}>{m.display_name}</span>
                )}
              </div>
            ))}
          </div>
        )}

        {/* Task attach popup */}
        {showAttach && (
          <TaskAttachPopup
            list={list}
            onSelect={(task) => { setAttachedTask(task); setShowAttach(false); }}
            onCreateNew={() => { setShowAttach(false); setShowTaskForm(true); }}
            onClose={() => setShowAttach(false)}
          />
        )}

        {/* Task form modal for creating new task */}
        {showTaskForm && (
          <TaskFormModal
            task={null}
            onSave={async (form) => {
              try {
                const newTask = await api.post("/api/tasks", { ...form, list_id: list.id });
                setAttachedTask(newTask);
                setShowTaskForm(false);
              } catch (err) { showToast(err.message, "error"); }
            }}
            onOfflineSave={() => setShowTaskForm(false)}
            onClose={() => setShowTaskForm(false)}
            sharedLists={[list]}
            collaborators={[]}
            projects={[]}
            contexts={[]}
          />
        )}

        {/* Attached task preview */}
        {attachedTask && (
          <div style={{ position: "absolute", bottom: 64, left: 12, background: "var(--bg-card)", border: "1px solid var(--accent)", borderRadius: 8, padding: "6px 10px", display: "flex", alignItems: "center", gap: 8, fontSize: 13 }}>
            <span style={{ color: "var(--accent)" }}>📌</span>
            <span style={{ fontWeight: 600, color: "var(--text-primary)" }}>{attachedTask.title}</span>
            <button onClick={() => setAttachedTask(null)} style={{ background: "none", border: "none", cursor: "pointer", color: "var(--text-light)", fontSize: 14, padding: "0 2px" }}>✕</button>
          </div>
        )}

        <textarea
          ref={textareaRef}
          className="chat-input"
          rows={1}
          placeholder="Ketik pesan... (Shift+Enter untuk baris baru, @ untuk mention)"
          value={text}
          onChange={handleTextChange}
          onKeyDown={handleKeyDown}
          style={{ marginBottom: attachedTask ? 36 : 0 }}
        />
        <button
          title="Attach atau buat task"
          onClick={() => setShowAttach(v => !v)}
          style={{ background: showAttach ? "var(--accent)" : "var(--bg-primary)", color: showAttach ? "white" : "var(--text-secondary)", border: "1px solid var(--border)", borderRadius: 8, padding: "8px 10px", cursor: "pointer", fontSize: 16, flexShrink: 0 }}>
          📌
        </button>
        <button className="btn btn-primary btn-sm" onClick={handleSend} disabled={sending || (!text.trim() && !attachedTask)} style={{ flexShrink: 0 }}>
          {sending ? "..." : "Kirim"}
        </button>
      </div>
    );
  }
